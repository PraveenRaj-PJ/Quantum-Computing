📋 Google Apps Script Code 
const SHEET_NAME = "Scores";
const ADMIN_PASS = "QuantumAdmin2025";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName(SHEET_NAME);
    if (!sheet) {
      sheet = ss.insertSheet(SHEET_NAME);
      sheet.appendRow(["Timestamp","Name","Email","Level",
                       "Score","Total","Percentage","Time(s)","Date"]);
      sheet.getRange(1,1,1,9).setFontWeight("bold");
    }
    if (data.action === "submit") {
      sheet.appendRow([
        new Date().toISOString(), data.name, data.email,
        data.level, data.score, data.total,
        data.pct + "%", data.timeTaken, data.date
      ]);
      return resp({ok:true});
    }
    if (data.action === "getScores" && data.pass === ADMIN_PASS) {
      const rows = sheet.getDataRange().getValues();
      return resp({ok:true, rows:rows});
    }
    if (data.action === "clearScores" && data.pass === ADMIN_PASS) {
      const hdr = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues();
      sheet.clearContents();
      sheet.getRange(1,1,1,hdr[0].length).setValues(hdr);
      sheet.getRange(1,1,1,hdr[0].length).setFontWeight("bold");
      return resp({ok:true});
    }
    return resp({ok:false, msg:"Unauthorized"});
  } catch(err) {
    return resp({ok:false, msg:err.toString()});
  }
}

function resp(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
